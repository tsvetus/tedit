class Test extends React.Component {
    constructor() {
        super();
        this.state = {value: 1}
    }
    render() {
        return (
            <TCheck
                label={'Check me:'}
                value={this.state.value}
                checked={1}
                unchecked={0}
                onChange={(event) => {this.setState({value: event.value});}} />
        );
    }
}
